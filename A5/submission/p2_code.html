Classifying the critical points found using the Second Derivative Test

In[]:
def classify(row: pd.DataFrame):
    if row["D(x,y)"] < 0:
        return "saddle"
    elif row["D(x,y)"] == 0:
        return "unknown"
    else:
        if row["f_xx"] > 0:
            return "minimum"
        else:
            return "maximum"
        
        
crit["classification"] = crit.apply(classify, axis = 1)
